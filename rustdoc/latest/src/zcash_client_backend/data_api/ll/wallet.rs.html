71impl<AccountId> WalletTransparentOutputs<AccountId> {
72    fn empty() -> Self {
73        Self {
74            #[cfg(feature = "transparent-inputs")]
75            received: vec![],
76            sent: vec![],
77        }
78    }
79
80    fn is_empty(&self) -> bool {
81        #[cfg(feature = "transparent-inputs")]
82        let has_received = !self.received.is_empty();
83        #[cfg(not(feature = "transparent-inputs"))]
84        let has_received = false;
85
86        let has_sent = !self.sent.is_empty();
87
88        !(has_received || has_sent)
89    }
90}

92pub(crate) fn determine_fee<DbT, T: TxMeta>(
93    _wallet_db: &DbT,
94    tx: &T,
95) -> Result<Option<Zatoshis>, DbT::Error>
96where
97    DbT: LowLevelWalletRead,
98    DbT::Error: From<BalanceError>,
99{
100    tx.fee_paid(|_outpoint| {
101        #[cfg(not(feature = "transparent-inputs"))]
102        {
103            // Transparent inputs aren't supported, so this closure should never be
104            // called during transaction construction. But in case it is, handle it
105            // correctly.
106            Ok(None)
107        }
108
109        // This closure can do DB lookups to fetch the value of each transparent input.
110        #[cfg(feature = "transparent-inputs")]
111        if let Some(out) = _wallet_db.get_wallet_transparent_output(_outpoint, None)? {
112            Ok(Some(out.txout().value()))
113        } else {
114            // If we canâ€™t find it, fee computation can't complete accurately
115            Ok(None)
116        }
117    })
118}

120/// Generates transparent gap addresses for a given account and key scope.
121///
122/// This is a convenience function that resolves the account's viewing keys from the wallet
123/// database and delegates to [`generate_gap_addresses`].
124#[cfg(feature = "transparent-inputs")]
125pub fn generate_transparent_gap_addresses<DbT, SE>(
126    wallet_db: &mut DbT,
127    gap_limits: GapLimits,
128    account_id: <DbT as LowLevelWalletRead>::AccountId,
129    key_scope: TransparentKeyScope,
130    request: UnifiedAddressRequest,
131) -> Result<(), GapAddressesError<SE>>
132where
133    DbT: LowLevelWalletWrite<Error = SE>
134        + AddressStore<Error = SE, AccountRef = <DbT as LowLevelWalletRead>::AccountRef>,
135    DbT::TxRef: Eq + Hash,
136{
137    let account_ref = wallet_db
138        .get_account_ref(account_id)

140
141    let account = wallet_db
142        .get_account_internal(account_ref)
143        .map_err(GapAddressesError::Storage)?
144        .ok_or(GapAddressesError::AccountUnknown)?;
145
146    generate_gap_addresses(
147        wallet_db,
148        &gap_limits,
149        account_ref,
150        &account.uivk(),
151        account.ufvk(),
152        key_scope,
153        request,
154        false,
155    )?;
156
157    Ok(())

229/// Adds information about a sequence of scanned blocks to the provided data store.
230///
231/// # Parameters
232/// - `wallet_db`: A handle to the underlying data store.
233/// - `from_state`: The note commitment tree state as of the end of the last block prior to the
234///   first block in the provided block vector.
235/// - `blocks`: The scanned block data to be added to the data store. This vector must contain
236///   data for blocks in sequentially increasing height order;
237///   [`PutBlocksError::NonSequentialBlocks`] will be returned if this invariant is violated.
238pub fn put_blocks<DbT, SE, TE>(
239    wallet_db: &mut DbT,
240    #[cfg(feature = "transparent-inputs")] gap_limits: GapLimits,
241    from_state: &ChainState,
242    blocks: Vec<ScannedBlock<<DbT as LowLevelWalletRead>::AccountId>>,
243) -> Result<(), PutBlocksError<SE, TE>>
244where
245    DbT: PutBlocksDbT<SE, TE, <DbT as LowLevelWalletRead>::AccountRef>,
246    DbT::TxRef: Eq + Hash,
247{
248    struct BlockPositions {
249        height: BlockHeight,
250        sapling_start_position: Position,
251        #[cfg(feature = "orchard")]
252        orchard_start_position: Position,
253    }
254
255    if blocks.is_empty() {
256        return Ok(());
257    }
258
259    let initial_block = blocks.first().expect("blocks is known to be nonempty");
260    assert!(from_state.block_height() + 1 == initial_block.height());
261
262    let start_positions = BlockPositions {
263        height: initial_block.height(),
264        sapling_start_position: Position::from(
265            u64::from(initial_block.sapling().final_tree_size())
266                - u64::try_from(initial_block.sapling().commitments().len()).unwrap(),
267        ),
268        #[cfg(feature = "orchard")]
269        orchard_start_position: Position::from(
270            u64::from(initial_block.orchard().final_tree_size())
271                - u64::try_from(initial_block.orchard().commitments().len()).unwrap(),
272        ),
273    };
274
275    let mut sapling_commitments = vec![];
276    #[cfg(feature = "orchard")]
277    let mut orchard_commitments = vec![];
278    let mut last_scanned_height = None;
279    let mut note_positions = vec![];
280
281    #[cfg(feature = "transparent-inputs")]
282    let mut tx_refs = HashSet::new();
283
284    for block in blocks.into_iter() {
285        if last_scanned_height
286            .iter()
287            .any(|prev| block.height() != *prev + 1)
288        {
289            return Err(PutBlocksError::NonSequentialBlocks {
290                prev_height: last_scanned_height.expect("last scanned height is known"),
291                block_height: block.height(),
292            });
293        }
294
295        // Insert the block into the database.
296        wallet_db
297            .put_block_meta(
298                block.height(),
299                block.block_hash(),
300                block.block_time(),
301                block.sapling().final_tree_size(),
302                block.sapling().commitments().len().try_into().unwrap(),
303                #[cfg(feature = "orchard")]
304                block.orchard().final_tree_size(),
305                #[cfg(feature = "orchard")]
306                block.orchard().commitments().len().try_into().unwrap(),
307            )
308            .map_err(PutBlocksError::Storage)?;
309
310        for tx in block.transactions() {
311            let tx_ref = wallet_db
312                .put_tx_meta(tx, block.height())
313                .map_err(PutBlocksError::Storage)?;
314
315            #[cfg(feature = "transparent-inputs")]
316            tx_refs.insert(tx_ref);
317
318            wallet_db
319                .queue_tx_retrieval(std::iter::once(tx.txid()), None)
320                .map_err(PutBlocksError::Storage)?;
321
322            // Mark notes as spent and remove them from the scanning cache
323            for spend in tx.sapling_spends() {
324                wallet_db
325                    .mark_sapling_note_spent(spend.nf(), tx_ref)
326                    .map_err(PutBlocksError::Storage)?;
327            }
328            #[cfg(feature = "orchard")]
329            for spend in tx.orchard_spends() {
330                wallet_db
331                    .mark_orchard_note_spent(spend.nf(), tx_ref)
332                    .map_err(PutBlocksError::Storage)?;
333            }
334
335            for output in tx.sapling_outputs() {
336                // Check whether this note was spent in a later block range that
337                // we previously scanned.
338                let spent_in = output
339                    .nf()
340                    .map(|nf| wallet_db.detect_sapling_spend(nf))
341                    .transpose()
342                    .map_err(PutBlocksError::Storage)?
343                    .flatten();
344
345                wallet_db
346                    .put_received_sapling_note(output, tx_ref, Some(block.height()), spent_in)
347                    .map_err(PutBlocksError::Storage)?;
348            }
349            #[cfg(feature = "orchard")]
350            for output in tx.orchard_outputs() {
351                // Check whether this note was spent in a later block range that
352                // we previously scanned.
353                let spent_in = output
354                    .nf()
355                    .map(|nf| wallet_db.detect_orchard_spend(nf))
356                    .transpose()
357                    .map_err(PutBlocksError::Storage)?
358                    .flatten();
359
360                wallet_db
361                    .put_received_orchard_note(output, tx_ref, Some(block.height()), spent_in)
362                    .map_err(PutBlocksError::Storage)?;
363            }
364        }
365
366        // Insert the new nullifiers from this block into the nullifier map.
367        wallet_db
368            .track_block_sapling_nullifiers(block.height(), block.sapling().nullifier_map())
369            .map_err(PutBlocksError::Storage)?;
370
371        #[cfg(feature = "orchard")]
372        wallet_db
373            .track_block_orchard_nullifiers(block.height(), block.orchard().nullifier_map())
374            .map_err(PutBlocksError::Storage)?;
375
376        note_positions.extend(block.transactions().iter().flat_map(|wtx| {
377            let iter = wtx.sapling_outputs().iter().map(|out| {
378                (
379                    ShieldedProtocol::Sapling,
380                    out.note_commitment_tree_position(),
381                )
382            });
383            #[cfg(feature = "orchard")]
384            let iter = iter.chain(wtx.orchard_outputs().iter().map(|out| {
385                (
386                    ShieldedProtocol::Orchard,
387                    out.note_commitment_tree_position(),
388                )
389            }));
390
391            iter
392        }));
393
394        last_scanned_height = Some(block.height());
395        let block_commitments = block.into_commitments();
396        trace!(
397            "Sapling commitments for {:?}: {:?}",
398            last_scanned_height,
399            block_commitments
400                .sapling
401                .iter()
402                .map(|(_, r)| *r)
403                .collect::<Vec<_>>()
404        );
405        #[cfg(feature = "orchard")]
406        trace!(
407            "Orchard commitments for {:?}: {:?}",
408            last_scanned_height,
409            block_commitments
410                .orchard
411                .iter()
412                .map(|(_, r)| *r)
413                .collect::<Vec<_>>()
414        );
415
416        sapling_commitments.extend(block_commitments.sapling.into_iter().map(Some));
417        #[cfg(feature = "orchard")]
418        orchard_commitments.extend(block_commitments.orchard.into_iter().map(Some));
419    }
420
421    #[cfg(feature = "transparent-inputs")]
422    for (account_id, key_scope) in wallet_db
423        .find_involved_accounts(tx_refs)
424        .map_err(PutBlocksError::Storage)?
425    {
426        if let Some(t_key_scope) = key_scope {
427            use ReceiverRequirement::*;
428            generate_transparent_gap_addresses(
429                wallet_db,
430                gap_limits,
431                account_id,
432                t_key_scope,
433                UnifiedAddressRequest::unsafe_custom(Allow, Allow, Require),
434            )
435            .map_err(PutBlocksError::GapAddresses)?;
436        }
437    }
438
439    // Prune the nullifier map of entries we no longer need.
440    wallet_db
441        .prune_tracked_nullifiers(PRUNING_DEPTH)
442        .map_err(PutBlocksError::Storage)?;
443
444    // We will have a start position and a last scanned height in all cases where
445    // `blocks` is non-empty.
446    if let Some(last_scanned_height) = last_scanned_height {
447        // Create subtrees from the note commitments in parallel.
448        const CHUNK_SIZE: usize = 1024;
449        let sapling_subtrees = sapling_commitments
450            .par_chunks_mut(CHUNK_SIZE)
451            .enumerate()
452            .filter_map(|(i, chunk)| {
453                let start = start_positions.sapling_start_position + (i * CHUNK_SIZE) as u64;
454                let end = start + chunk.len() as u64;
455
456                shardtree::LocatedTree::from_iter(
457                    start..end,
458                    SAPLING_SHARD_HEIGHT.into(),
459                    chunk.iter_mut().map(|n| n.take().expect("always Some")),
460                )
461            })
462            .map(|res| (res.subtree, res.checkpoints))
463            .collect::<Vec<_>>();
464
465        #[cfg(feature = "orchard")]
466        let orchard_subtrees = orchard_commitments
467            .par_chunks_mut(CHUNK_SIZE)
468            .enumerate()
469            .filter_map(|(i, chunk)| {
470                let start = start_positions.orchard_start_position + (i * CHUNK_SIZE) as u64;
471                let end = start + chunk.len() as u64;
472
473                shardtree::LocatedTree::from_iter(
474                    start..end,
475                    ORCHARD_SHARD_HEIGHT.into(),
476                    chunk.iter_mut().map(|n| n.take().expect("always Some")),
477                )
478            })
479            .map(|res| (res.subtree, res.checkpoints))
480            .collect::<Vec<_>>();
481
482        // Collect the complete set of Sapling checkpoints
483        #[cfg(feature = "orchard")]
484        let sapling_checkpoint_positions: BTreeMap<BlockHeight, Position> = sapling_subtrees
485            .iter()
486            .flat_map(|(_, checkpoints)| checkpoints.iter())
487            .map(|(k, v)| (*k, *v))
488            .collect();
489
490        #[cfg(feature = "orchard")]
491        let orchard_checkpoint_positions: BTreeMap<BlockHeight, Position> = orchard_subtrees
492            .iter()
493            .flat_map(|(_, checkpoints)| checkpoints.iter())
494            .map(|(k, v)| (*k, *v))
495            .collect();
496
497        #[cfg(feature = "orchard")]
498        fn ensure_checkpoints<'a, H, I: Iterator<Item = &'a BlockHeight>, const DEPTH: u8>(
499            // An iterator of checkpoints heights for which we wish to ensure that
500            // checkpoints exists.
501            ensure_heights: I,
502            // The map of checkpoint positions from which we will draw note commitment tree
503            // position information for the newly created checkpoints.
504            existing_checkpoint_positions: &BTreeMap<BlockHeight, Position>,
505            // The frontier whose position will be used for an inserted checkpoint when
506            // there is no preceding checkpoint in existing_checkpoint_positions.
507            state_final_tree: &Frontier<H, DEPTH>,
508        ) -> Vec<(BlockHeight, Checkpoint)> {
509            ensure_heights
510                .flat_map(|ensure_height| {
511                    existing_checkpoint_positions
512                        .range::<BlockHeight, _>(..=*ensure_height)
513                        .last()
514                        .map_or_else(
515                            || {
516                                Some((
517                                    *ensure_height,
518                                    state_final_tree
519                                        .value()
520                                        .map_or_else(Checkpoint::tree_empty, |t| {
521                                            Checkpoint::at_position(t.position())
522                                        }),
523                                ))
524                            },
525                            |(existing_checkpoint_height, position)| {
526                                if *existing_checkpoint_height < *ensure_height {
527                                    Some((*ensure_height, Checkpoint::at_position(*position)))
528                                } else {
529                                    // The checkpoint already exists, so we don't need to
530                                    // do anything.
531                                    None
532                                }
533                            },
534                        )
535                        .into_iter()
536                })
537                .collect::<Vec<_>>()
538        }
539
540        #[cfg(feature = "orchard")]
541        let (missing_sapling_checkpoints, missing_orchard_checkpoints) = (
542            ensure_checkpoints(
543                orchard_checkpoint_positions.keys(),
544                &sapling_checkpoint_positions,
545                from_state.final_sapling_tree(),
546            ),
547            ensure_checkpoints(
548                sapling_checkpoint_positions.keys(),
549                &orchard_checkpoint_positions,
550                from_state.final_orchard_tree(),
551            ),
552        );
553
554        // Update the Sapling note commitment tree with all newly read note commitments
555        {
556            let mut sapling_subtrees_iter = sapling_subtrees.into_iter();
557            wallet_db.with_sapling_tree_mut::<_, _, PutBlocksError<SE, TE>>(|sapling_tree| {
558                debug!(
559                    "Sapling initial tree size at {:?}: {:?}",
560                    from_state.block_height(),
561                    from_state.final_sapling_tree().tree_size()
562                );
563                // We insert the frontier with `Checkpoint` retention because we need to be
564                // able to truncate the tree back to this point.
565                sapling_tree.insert_frontier(
566                    from_state.final_sapling_tree().clone(),
567                    Retention::Checkpoint {
568                        id: from_state.block_height(),
569                        marking: Marking::Reference,
570                    },
571                )?;
572
573                for (tree, checkpoints) in &mut sapling_subtrees_iter {
574                    sapling_tree.insert_tree(tree, checkpoints)?;
575                }
576
577                // Ensure we have a Sapling checkpoint for each checkpointed Orchard block height.
578                // We skip all checkpoints below the minimum retained checkpoint in the
579                // Sapling tree, because branches below this height may be pruned.
580                #[cfg(feature = "orchard")]
581                {
582                    let min_checkpoint_height = sapling_tree
583                        .store()
584                        .min_checkpoint_id()
585                        .map_err(ShardTreeError::Storage)?
586                        .expect("At least one checkpoint was inserted (by insert_frontier)");
587
588                    for (height, checkpoint) in &missing_sapling_checkpoints {
589                        if *height > min_checkpoint_height {
590                            sapling_tree
591                                .store_mut()
592                                .add_checkpoint(*height, checkpoint.clone())
593                                .map_err(ShardTreeError::Storage)?;
594                        }
595                    }
596                }
597
598                Ok(())
599            })?;
600        }
601
602        // Update the Orchard note commitment tree with all newly read note commitments
603        #[cfg(feature = "orchard")]
604        {
605            let mut orchard_subtrees = orchard_subtrees.into_iter();
606            wallet_db.with_orchard_tree_mut::<_, _, PutBlocksError<SE, TE>>(|orchard_tree| {
607                debug!(
608                    "Orchard initial tree size at {:?}: {:?}",
609                    from_state.block_height(),
610                    from_state.final_orchard_tree().tree_size()
611                );
612                // We insert the frontier with `Checkpoint` retention because we need to be
613                // able to truncate the tree back to this point.
614                orchard_tree.insert_frontier(
615                    from_state.final_orchard_tree().clone(),
616                    Retention::Checkpoint {
617                        id: from_state.block_height(),
618                        marking: Marking::Reference,
619                    },
620                )?;
621
622                for (tree, checkpoints) in &mut orchard_subtrees {
623                    orchard_tree.insert_tree(tree, checkpoints)?;
624                }
625
626                // Ensure we have an Orchard checkpoint for each checkpointed Sapling block height.
627                // We skip all checkpoints below the minimum retained checkpoint in the
628                // Orchard tree, because branches below this height may be pruned.
629                {
630                    let min_checkpoint_height = orchard_tree
631                        .store()
632                        .min_checkpoint_id()
633                        .map_err(ShardTreeError::Storage)?
634                        .expect("At least one checkpoint was inserted (by insert_frontier)");
635
636                    for (height, checkpoint) in &missing_orchard_checkpoints {
637                        if *height > min_checkpoint_height {
638                            debug!(
639                                "Adding missing Orchard checkpoint for height: {:?}: {:?}",
640                                height,
641                                checkpoint.position()
642                            );
643                            orchard_tree
644                                .store_mut()
645                                .add_checkpoint(*height, checkpoint.clone())
646                                .map_err(ShardTreeError::Storage)?;
647                        }
648                    }
649                }
650                Ok(())
651            })?;
652        }
653
654        wallet_db
655            .notify_scan_complete(
656                Range {
657                    start: start_positions.height,
658                    end: last_scanned_height + 1,
659                },
660                &note_positions,
661            )
662            .map_err(PutBlocksError::Storage)?;
663    }
664
665    Ok(())
666}

715/// Persists a decrypted transaction to the wallet database.
716///
717/// This function stores a transaction that has been decrypted by the wallet, including:
718/// - The transaction data and any computed fee (if all inputs are known)
719/// - Received shielded notes (Sapling and Orchard)
720/// - Sent outputs with recipient information
721/// - Transparent outputs received by or sent from the wallet
722/// - Nullifier tracking for spent notes
723///
724/// The function also queues requests for retrieval of any unknown transparent inputs,
725/// which may be needed to compute the transaction fee or track wallet history.
726///
727/// # Parameters
728/// - `wallet_db`: The wallet database to update.
729/// - `params`: The network parameters.
730/// - `chain_tip_height`: The current chain tip height, used as the observation height for
731///   unmined transactions.
732/// - `d_tx`: The decrypted transaction to store.
733///
734/// # Returns
735/// Returns `Ok(())` if the transaction was successfully stored, or an error if a database
736/// operation failed.
737pub fn store_decrypted_tx<DbT, P>(
738    wallet_db: &mut DbT,
739    params: &P,
740    #[cfg(feature = "transparent-inputs")] gap_limits: GapLimits,
741    chain_tip_height: BlockHeight,
742    d_tx: DecryptedTransaction<Transaction, <DbT as LowLevelWalletRead>::AccountId>,
743) -> Result<(), <DbT as LowLevelWalletRead>::Error>
744where
745    DbT: StoreDecryptedTxDbT,
746    <DbT as LowLevelWalletRead>::AccountId: core::fmt::Debug,
747    <DbT as LowLevelWalletRead>::Error: From<BalanceError> + From<GapError<DbT>>,
748    P: consensus::Parameters,
749{
750    let funding_accounts = wallet_db.get_funding_accounts(d_tx.tx())?;
751
752    // TODO(#1305): Correctly track accounts that fund each transaction output.
753    let funding_account = funding_accounts.iter().next().copied();
754    if funding_accounts.len() > 1 {
755        warn!(
756            "More than one wallet account detected as funding transaction {:?}, selecting {:?}",
757            d_tx.tx().txid(),
758            funding_account.unwrap()
759        )
760    }
761
762    let wallet_transparent_outputs = detect_wallet_transparent_outputs::<DbT, P>(
763        #[cfg(feature = "transparent-inputs")]
764        wallet_db,
765        params,
766        &d_tx,
767        funding_account,
768    )?;
769
770    // If there is no wallet involvement, we don't need to store the transaction, so just return
771    // here.
772    if funding_account.is_none()
773        && wallet_transparent_outputs.is_empty()
774        && !d_tx.has_decrypted_outputs()
775    {
776        wallet_db.delete_retrieval_queue_entries(d_tx.tx().txid())?;
777        return Ok(());
778    }
779
780    info!("Storing decrypted transaction with id {}", d_tx.tx().txid());
781    let observed_height = d_tx.mined_height().unwrap_or(chain_tip_height + 1);
782
783    // If the transaction is fully shielded, or all transparent inputs are available, set the
784    // fee value.
785    let fee = determine_fee(wallet_db, d_tx.tx())?;
786
787    let tx_ref = wallet_db.put_tx_data(d_tx.tx(), fee, None, None, observed_height)?;
788    if let Some(height) = d_tx.mined_height() {
789        wallet_db.set_transaction_status(d_tx.tx().txid(), TransactionStatus::Mined(height))?;
790    }
791
792    // Mark Sapling notes as spent when we observe their nullifiers.
793    for spend in d_tx
794        .tx()
795        .sapling_bundle()
796        .iter()
797        .flat_map(|b| b.shielded_spends().iter())
798    {
799        wallet_db.mark_sapling_note_spent(spend.nullifier(), tx_ref)?;
800    }
801
802    // Mark Orchard notes as spent when we observe their nullifiers.
803    #[cfg(feature = "orchard")]
804    for action in d_tx
805        .tx()
806        .orchard_bundle()
807        .iter()
808        .flat_map(|b| b.actions().iter())
809    {
810        wallet_db.mark_orchard_note_spent(action.nullifier(), tx_ref)?;
811    }
812
813    // If any of the utxos spent in the transaction are ours, mark them as spent.
814    #[cfg(feature = "transparent-inputs")]
815    for txin in d_tx
816        .tx()
817        .transparent_bundle()
818        .iter()
819        .flat_map(|b| b.vin.iter())
820    {
821        wallet_db.mark_transparent_utxo_spent(txin.prevout(), tx_ref)?;
822    }
823
824    // A flag used to determine whether it is necessary to query for transactions that
825    // provided transparent inputs to this transaction, in order to be able to correctly
826    // recover transparent transaction history.
827    #[cfg(feature = "transparent-inputs")]
828    let mut tx_has_wallet_outputs = false;
829    #[cfg(feature = "transparent-inputs")]
830    {
831        tx_has_wallet_outputs |= !d_tx.sapling_outputs().is_empty();
832
833        #[cfg(feature = "orchard")]
834        {
835            tx_has_wallet_outputs |= !d_tx.orchard_outputs().is_empty();
836        }
837
838        // Since the wallet created the transparent output, we need to ensure
839        // that any transparent inputs belonging to the wallet will be
840        // discovered.
841        tx_has_wallet_outputs |= !wallet_transparent_outputs.received.is_empty();
842
843        // Even though we know the funding account, we don't know that we have
844        // information for all of the transparent inputs to the transaction.
845        tx_has_wallet_outputs |= !wallet_transparent_outputs.sent.is_empty();
846    }
847
848    // The set of account/scope pairs for which to update the gap limit.
849    #[cfg(feature = "transparent-inputs")]
850    let mut gap_update_set = HashSet::new();
851
852    put_shielded_outputs(
853        wallet_db,
854        params,
855        tx_ref,
856        funding_account,
857        d_tx.sapling_outputs(),
858        PoolType::SAPLING,
859        Note::Sapling,
860        |note| Receiver::Sapling(note.recipient()),
861        |output| output.note_value(),
862        |wallet_db, output, tx_ref| {
863            wallet_db.put_received_sapling_note(output, tx_ref, d_tx.mined_height(), None)
864        },
865        |_account_id| {
866            #[cfg(feature = "transparent-inputs")]
867            gap_update_set.insert((_account_id, TransparentKeyScope::EXTERNAL));
868        },
869    )?;
870
871    #[cfg(feature = "orchard")]
872    put_shielded_outputs(
873        wallet_db,
874        params,
875        tx_ref,
876        funding_account,
877        d_tx.orchard_outputs(),
878        PoolType::ORCHARD,
879        Note::Orchard,
880        |note| Receiver::Orchard(note.recipient()),
881        |output| output.note_value(),
882        |wallet_db, output, tx_ref| {
883            wallet_db.put_received_orchard_note(output, tx_ref, d_tx.mined_height(), None)
884        },
885        |_account_id| {
886            #[cfg(feature = "transparent-inputs")]
887            gap_update_set.insert((_account_id, TransparentKeyScope::EXTERNAL));
888        },
889    )?;
890
891    put_transparent_outputs(
892        wallet_db,
893        tx_ref,
894        &wallet_transparent_outputs,
895        #[cfg(feature = "transparent-inputs")]
896        |wallet_db, output| wallet_db.put_transparent_output(output, observed_height, false),
897        #[cfg(feature = "transparent-inputs")]
898        |account_id, t_key_scope| {
899            gap_update_set.insert((account_id, t_key_scope));
900        },
901    )?;
902
903    // Regenerate the gap limit addresses.
904    #[cfg(feature = "transparent-inputs")]
905    for (account_id, key_scope) in gap_update_set {
906        use ReceiverRequirement::*;
907        generate_transparent_gap_addresses(
908            wallet_db,
909            gap_limits,
910            account_id,
911            key_scope,
912            UnifiedAddressRequest::unsafe_custom(Allow, Allow, Require),
913        )?;
914    }
915
916    // For each transaction that spends a transparent output of this transaction and does not
917    // already have a known fee value, set the fee if possible.
918    for (spending_tx_ref, spending_tx) in
919        wallet_db.get_txs_spending_transparent_outputs_of(tx_ref)?
920    {
921        if let Some(fee) = determine_fee(wallet_db, &spending_tx)? {
922            wallet_db.update_tx_fee(spending_tx_ref, fee)?;
923        }
924    }
925
926    // If the transaction has outputs that belong to the wallet as well as transparent
927    // inputs, we may need to download the transactions corresponding to the transparent
928    // prevout references to determine whether the transaction was created (at least in
929    // part) by this wallet.
930    #[cfg(feature = "transparent-inputs")]
931    if tx_has_wallet_outputs {
932        wallet_db.queue_transparent_input_retrieval(tx_ref, &d_tx)?
933    }
934
935    wallet_db.delete_retrieval_queue_entries(d_tx.tx().txid())?;
936
937    // If the decrypted transaction is unmined and has no shielded components, add it to
938    // the queue for status retrieval.
939    #[cfg(feature = "transparent-inputs")]
940    {
941        let detectable_via_scanning = d_tx.tx().sapling_bundle().is_some();
942        #[cfg(feature = "orchard")]
943        let detectable_via_scanning =
944            detectable_via_scanning | d_tx.tx().orchard_bundle().is_some();
945
946        if d_tx.mined_height().is_none() && !detectable_via_scanning {
947            wallet_db.queue_tx_retrieval(std::iter::once(d_tx.tx().txid()), None)?
948        }
949    }
950
951    Ok(())
952}

954fn detect_wallet_transparent_outputs<DbT, P>(
955    #[cfg(feature = "transparent-inputs")] wallet_db: &DbT,
956    params: &P,
957    d_tx: &DecryptedTransaction<Transaction, DbT::AccountId>,
958    funding_account: Option<DbT::AccountId>,
959) -> Result<WalletTransparentOutputs<DbT::AccountId>, DbT::Error>
960where
961    DbT: LowLevelWalletRead,
962    DbT::AccountId: core::fmt::Debug,
963    P: consensus::Parameters,
964{
965    // This `if` is just an optimization for cases where we would do nothing in the loop.
966    if funding_account.is_some() || cfg!(feature = "transparent-inputs") {
967        let mut result = WalletTransparentOutputs::empty();
968        for (output_index, txout) in d_tx
969            .tx()
970            .transparent_bundle()
971            .iter()
972            .flat_map(|b| b.vout.iter())
973            .enumerate()
974        {
975            if let Some(address) = txout.recipient_address() {
976                debug!(
977                    "{:?} output {} has recipient {}",
978                    d_tx.tx().txid(),
979                    output_index,
980                    address.encode(params)
981                );
982
983                // If the output belongs to the wallet, add it to `transparent_received_outputs`.
984                #[cfg(feature = "transparent-inputs")]
985                if let Some((account_uuid, key_scope)) =
986                    wallet_db.find_account_for_transparent_address(&address)?
987                {
988                    debug!(
989                        "{:?} output {} belongs to account {:?}",
990                        d_tx.tx().txid(),
991                        output_index,
992                        account_uuid
993                    );
994                    result.received.push((
995                        WalletTransparentOutput::from_parts(
996                            OutPoint::new(
997                                d_tx.tx().txid().into(),
998                                u32::try_from(output_index).unwrap(),
999                            ),
1000                            txout.clone(),
1001                            d_tx.mined_height(),
1002                        )
1003                        .expect("txout.recipient_address extraction previously checked"),
1004                        key_scope,
1005                    ));
1006                } else {
1007                    debug!(
1008                        "Address {} is not recognized as belonging to any of our accounts.",
1009                        address.encode(params)
1010                    );
1011                }
1012
1013                // If a transaction we observe contains spends from our wallet, we will
1014                // store its transparent outputs in the same way they would be stored by
1015                // create_spend_to_address.
1016                if let Some(account_uuid) = funding_account {
1017                    let receiver = Receiver::Transparent(address);
1018
1019                    #[cfg(feature = "transparent-inputs")]
1020                    let recipient_address = wallet_db
1021                        .select_receiving_address(account_uuid, &receiver)?
1022                        .unwrap_or_else(|| receiver.to_zcash_address(params.network_type()));
1023
1024                    #[cfg(not(feature = "transparent-inputs"))]
1025                    let recipient_address = receiver.to_zcash_address(params.network_type());
1026
1027                    let recipient = Recipient::External {
1028                        recipient_address,
1029                        output_pool: PoolType::TRANSPARENT,
1030                    };
1031
1032                    result.sent.push(TransparentSentOutput {
1033                        from_account_uuid: account_uuid,
1034                        output_index,
1035                        recipient,
1036                        value: txout.value(),
1037                    });
1038                }
1039            } else {
1040                warn!(
1041                    "Unable to determine recipient address for tx {} output {}",
1042                    d_tx.tx().txid(),
1043                    output_index
1044                );
1045            }
1046        }
1047
1048        Ok(result)
1049    } else {
1050        Ok(WalletTransparentOutputs::empty())
1051    }
1052}

1054#[allow(clippy::too_many_arguments)]
1055fn put_shielded_outputs<DbT, P, N>(
1056    wallet_db: &mut DbT,
1057    params: &P,
1058    tx_ref: <DbT as LowLevelWalletRead>::TxRef,
1059    funding_account: Option<DbT::AccountId>,
1060    outputs: &[DecryptedOutput<N, <DbT as LowLevelWalletRead>::AccountId>],
1061    output_pool: PoolType,
1062    note: impl Fn(N) -> Note,
1063    receiver: impl Fn(&N) -> Receiver,
1064    value: impl Fn(&DecryptedOutput<N, <DbT as LowLevelWalletRead>::AccountId>) -> Zatoshis,
1065    put_received_note: impl Fn(
1066        &mut DbT,
1067        &DecryptedOutput<N, <DbT as LowLevelWalletRead>::AccountId>,
1068        <DbT as LowLevelWalletRead>::TxRef,
1069    ) -> Result<(), <DbT as LowLevelWalletRead>::Error>,
1070    mut on_external_account: impl FnMut(<DbT as LowLevelWalletRead>::AccountId),
1071) -> Result<(), <DbT as LowLevelWalletRead>::Error>
1072where
1073    DbT: LowLevelWalletWrite,
1074    P: consensus::Parameters,
1075    N: Clone,
1076{
1077    for output in outputs {
1078        match output.transfer_type() {
1079            TransferType::Outgoing => {
1080                let recipient = {
1081                    let receiver = receiver(output.note());
1082                    let recipient_address = wallet_db
1083                        .select_receiving_address(*output.account(), &receiver)?
1084                        .unwrap_or_else(|| receiver.to_zcash_address(params.network_type()));
1085
1086                    Recipient::External {
1087                        recipient_address,
1088                        output_pool,
1089                    }
1090                };
1091
1092                wallet_db.put_sent_output(
1093                    *output.account(),
1094                    tx_ref,
1095                    output.index(),
1096                    &recipient,
1097                    value(output),
1098                    Some(output.memo()),
1099                )?;
1100            }
1101            TransferType::WalletInternal => {
1102                put_received_note(wallet_db, output, tx_ref)?;
1103
1104                let recipient = Recipient::InternalAccount {
1105                    receiving_account: *output.account(),
1106                    external_address: None,
1107                    note: Box::new(note(output.note().clone())),
1108                };
1109
1110                wallet_db.put_sent_output(
1111                    *output.account(),
1112                    tx_ref,
1113                    output.index(),
1114                    &recipient,
1115                    value(output),
1116                    Some(output.memo()),
1117                )?;
1118            }
1119            TransferType::Incoming => {
1120                put_received_note(wallet_db, output, tx_ref)?;
1121                on_external_account(*output.account());
1122
1123                if let Some(account_id) = funding_account {
1124                    // Even if the recipient address is external, record the send as internal.
1125                    let recipient = Recipient::InternalAccount {
1126                        receiving_account: *output.account(),
1127                        external_address: {
1128                            let receiver = receiver(output.note());
1129                            Some(
1130                                wallet_db
1131                                    .select_receiving_address(*output.account(), &receiver)?
1132                                    .unwrap_or_else(|| {
1133                                        receiver.to_zcash_address(params.network_type())
1134                                    }),
1135                            )
1136                        },
1137                        note: Box::new(note(output.note().clone())),
1138                    };
1139
1140                    wallet_db.put_sent_output(
1141                        account_id,
1142                        tx_ref,
1143                        output.index(),
1144                        &recipient,
1145                        value(output),
1146                        Some(output.memo()),
1147                    )?;
1148                }
1149            }
1150        }
1151    }
1152
1153    Ok(())
1154}

1156fn put_transparent_outputs<DbT>(
1157    wallet_db: &mut DbT,
1158    tx_ref: <DbT as LowLevelWalletRead>::TxRef,
1159    outputs: &WalletTransparentOutputs<<DbT as LowLevelWalletRead>::AccountId>,
1160    #[cfg(feature = "transparent-inputs")] put_received_output: impl Fn(
1161        &mut DbT,
1162        &WalletTransparentOutput,
1163    ) -> Result<
1164        (
1165            <DbT as LowLevelWalletRead>::AccountId,
1166            std::option::Option<TransparentKeyScope>,
1167        ),
1168        <DbT as LowLevelWalletRead>::Error,
1169    >,
1170    #[cfg(feature = "transparent-inputs")] mut on_received: impl FnMut(
1171        <DbT as LowLevelWalletRead>::AccountId,
1172        TransparentKeyScope,
1173    ),
1174) -> Result<(), <DbT as LowLevelWalletRead>::Error>
1175where
1176    DbT: LowLevelWalletWrite,
1177{
1178    #[cfg(feature = "transparent-inputs")]
1179    for (received_t_output, key_scope) in &outputs.received {
1180        let (account_id, _) = put_received_output(wallet_db, received_t_output)?;
1181
1182        if let Some(t_key_scope) = key_scope {
1183            on_received(account_id, *t_key_scope);
1184        }
1185
1186        // When we receive transparent funds (particularly as ephemeral outputs
1187        // in transaction pairs sending to a ZIP 320 address) it becomes
1188        // possible that the spend of these outputs is not then later detected
1189        // if the transaction that spends them is purely transparent. This is
1190        // especially a problem in wallet recovery.
1191        wallet_db.queue_transparent_spend_detection(
1192            *received_t_output.recipient_address(),
1193            tx_ref,
1194            received_t_output.outpoint().n(),
1195        )?;
1196    }
1197
1198    for sent_t_output in &outputs.sent {
1199        wallet_db.put_sent_output(
1200            sent_t_output.from_account_uuid,
1201            tx_ref,
1202            sent_t_output.output_index,
1203            &sent_t_output.recipient,
1204            sent_t_output.value,
1205            None,
1206        )?;
1207    }
1208
1209    Ok(())
